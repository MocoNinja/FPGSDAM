<script language="JavaScript">
actual=1;
function cotejar(valor){
	alert(valor);
}
function pasar(){
	objeto = document.getElementById("pagina" + actual);	
	if((actual>1)&&((event.key == 'a')||(event.key == 'A'))){
		objeto.style.visibility = "hidden";
		actual--;
		objeto = document.getElementById("pagina" + actual);
		objeto.style.visibility = "inherit";
		}
	if((actual<6)&&((event.key == 's')||(event.key == 'S'))){
		objeto.style.visibility = "hidden";
		actual++;
		objeto = document.getElementById("pagina" + actual);
		objeto.style.visibility = "inherit";
		}
}
</script>
