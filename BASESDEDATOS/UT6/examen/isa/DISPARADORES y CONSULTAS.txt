TRIGGER 1.

DROP TRIGGER IF EXISTS controlAcceso //
CREATE TRIGGER controlAcceso
AFTER INSERT ON usuarios
for each row begin
 INSERT INTO bitacora(usuario,operacion,fecha) VALUES ( new.usuario, "Alta",now());
end
